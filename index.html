async function downloadVideo() { const urlInput = document.getElementById('videoUrl').value.trim(); if (!urlInput) return alert('Dán link vào đi b!'); // Thay vì fetch('https://caxaumc.pythonanywhere.com/get-video'), // b hãy dùng API này (nó không bị chặn mạng): const apiUrl = `https://api.tiklydown.eu.org/api/download?url=${encodeURIComponent(urlInput)}`; try { const response = await fetch(apiUrl); const data = await response.json(); if (data.status === 200 && data.video.noWatermark) { window.location.href = data.video.noWatermark; // Tải trực tiếp } else { alert('Lỗi rồi, để tui xem lại...'); } } catch (e) { alert('Lỗi kết nối: ' + e.message); } }
